149  for(int i=0; i<colisoes.size(); i++){
150  // Dados do primeiro Objeto da dupla de Colisão
151  v1 = move(colisoes[i].first->v);
152  r1 = move(colisoes[i].first->r);
153  m1 = colisoes[i].first->m;
154 
155  // Dados do segundo Objeto da dupla de Colisão
156  v2 = move(colisoes[i].second->v);
157  r2 = move(colisoes[i].second->r);
158  m2 = colisoes[i].second->m;
159 
160  // Cálculo do novo vetor Velocidade através do cálculo vetorial de colisão bidimensional entre dois Objetos
161  Vetor v1_ = v1 - ((2*m2)/(m1+m2)) * (((v1-v2)*(r1-r2)) * (1/pow((r1-r2).getModulo(),2))) * (r1-r2);
162  Vetor v2_ = v2 - ((2*m1)/(m1+m2)) * (((v2-v1)*(r2-r1)) * (1/pow((r2-r1).getModulo(),2))) * (r2-r1);
163

165  colisoes[i].first->v = move(v1_);

222  for (int j=0; j<qtd_linhas; j++){

226

228  float d;
229 
230  /*
231  Verificação se o Objeto colidiu com a Linha para cálculo da distância d,

233  */
234  if (j==0){
235  if (r.y + ob.raio_circunferencia >= l.p1.y){
236  colidiu = true;
237  d = r.y + raio - l.p1.y;
238  }
239  }
240  else if (j==1){
241  if (r.y - ob.raio_circunferencia <= l.p1.y){
242  colidiu = true;
243  d = abs(r.y - raio - l.p1.y);
244  }
245  }
246  else if(j==2){

248  colidiu = true;

250  }
251  }

253  if (r.x + ob.raio_circunferencia >= l.p1.x){
254  colidiu = true;
255  d = r.x + raio - l.p1.x;
256  }
257  }
258

260  if (colidiu){

262 
263 
264  // Cálculo do menor angulo formado entre o Vetor velocidade e o eixo das ordendas
265  if (angulo <= M_PI)
266  angulo = M_PI/2.f - angulo;
267  else
268  angulo = 3*M_PI/2.f - angulo;
269

271  if (sin(angulo) == sin(M_PI/2.f) || sin(angulo) == sin(-M_PI/2.f))
272  /*
273  Caso ocorra a igualdade anterior, deve-se considerar

278  angulo = 0;

281  if (j == 0){
282  (ptr_lista_objetos+i)->r.x -= d*tan(angulo);
283  (ptr_lista_objetos+i)->r.y -= d;
284  }

286  (ptr_lista_objetos+i)->r.x += d*tan(angulo);
